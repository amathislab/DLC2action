 21class MaskedFeaturesSSL(SSLConstructor, ABC):
 22    """
 23    A base masked features SSL class
 24
 25    Mask some of the input features randomly and predict the initial data.
 26    """
 27
 28    type = "ssl_input"
 29
 30    def __init__(self, frac_masked: float = 0.2) -> None:
 31        """
 32        Parameters
 33        ----------
 34        frac_masked : float
 35            fraction of features to real_lens
 36        """
 37
 38        super().__init__()
 39        self.mse = _MSE()
 40        self.frac_masked = frac_masked
 41
 42    def transformation(self, sample_data: Dict) -> Tuple:
 43        """
 44        Mask some of the features randomly
 45        """
 46
 47        for key in sample_data:
 48            mask = torch.empty(sample_data[key].shape).normal_() > self.frac_masked
 49            sample_data[key] = sample_data[key] * mask
 50        ssl_target = torch.cat(list(sample_data.values()))
 51        return (sample_data, ssl_target)
 52
 53    def loss(self, predicted: torch.Tensor, target: torch.Tensor) -> float:
 54        """
 55        MSE loss
 56        """
 57
 58        loss = self.mse(predicted, target)
 59        return loss

162class MaskedKinematicSSL(SSLConstructor, ABC):
163    """
164    A base masked joints SSL class
165
166    Mask some of the joints randomly and predict the initial data.
167    """
168
169    type = "ssl_input"
170
171    def __init__(self, frac_masked: float = 0.2) -> None:
172        """
173        Parameters
174        ----------
175        frac_masked : float, default 0.1
176            fraction of features to real_lens
177        """
178
179        super().__init__()
180        self.mse = _MSE()
181        self.frac_masked = frac_masked
182
183    def _get_keys(self, key_bases, x):
184        """
185        Get keys of x that start with one of the strings in key_bases
186        """
187
188        keys = []
189        for key in x:
190            if key.startswith(key_bases):
191                keys.append(key)
192        return keys
193
194    def transformation(self, sample_data: Dict) -> Tuple:
195        """
196        Mask joints randomly
197        """
198
199        key = self._get_keys("coords", sample_data)[0]
200        features, frames = sample_data[key].shape
201        n_bp = features // 2
202        masked_joints = torch.FloatTensor(n_bp).uniform_() > self.frac_masked
203        keys = self._get_keys(("intra_distance", "inter_distance"), sample_data)
204        for key in keys:
205            mask = masked_joints.repeat(n_bp, frames, 1).transpose(1, 2)
206            indices = torch.triu_indices(n_bp, n_bp, 1)
207
208            X = torch.zeros((n_bp, n_bp, frames)).to(sample_data[key].device)
209            X[indices[0], indices[1], :] = sample_data[key]
210            X[mask] = 0
211            X[mask.transpose(0, 1)] = 0
212            sample_data[key] = X[indices[0], indices[1], :].reshape(-1, frames)
213        keys = self._get_keys(("speed_joints", "coords", "acc_joints"), sample_data)
214        for key in keys:
215            mask = (
216                masked_joints.repeat(2, frames, 1).transpose(0, 2).reshape((-1, frames))
217            )
218            sample_data[key][mask] = 0
219        keys = self._get_keys("angle_joints_radian", sample_data)
220        for key in keys:
221            mask = masked_joints.repeat(frames, 1).transpose(0, 1)
222            sample_data[key][mask] = 0
223
224        return sample_data, torch.cat(list(sample_data.values()))
225
226    def loss(self, predicted: torch.Tensor, target: torch.Tensor) -> float:
227        """
228        MSE loss
229        """
230
231        loss = self.mse(predicted, target)
232        return loss

323class MaskedFramesSSL(SSLConstructor, ABC):
324    """Generates the functions necessary to build a masked features SSL: real_lens some of the input features randomly
325    and predict the initial data"""
326
327    type = "ssl_input"
328
329    def __init__(self, frac_masked: float = 0.1) -> None:
330        """
331        Parameters
332        ----------
333        frac_masked : float, default 0.1
334            fraction of frames to real_lens
335        """
336
337        super().__init__()
338        self.frac_masked = frac_masked
339        self.mse = _MSE()
340
341    def transformation(self, sample_data: Dict) -> Tuple:
342        """
343        Mask some of the frames randomly
344        """
345
346        key = list(sample_data.keys())[0]
347        num_frames = sample_data[key].shape[-1]
348        mask = torch.empty(num_frames).normal_() > self.frac_masked
349        mask = mask.unsqueeze(0)
350        for key in sample_data:
351            sample_data[key] = sample_data[key] * mask
352        ssl_target = torch.cat(list(sample_data.values()))
353        return (sample_data, ssl_target)
354
355    def loss(self, predicted: torch.Tensor, target: torch.Tensor) -> float:
356        """
357        MSE loss
358        """
359
360        loss = self.mse(predicted, target)
361        return loss

type = 'ssl_input'

The type parameter defines interaction with the model:

  • 'ssl_input': a modification of the input data passes through the base network feature extraction module and the SSL module; it is returned as SSL output and compared to SSL target (or, if it is None, to the input data),
  • 'ssl_output': the input data passes through the base network feature extraction module and the SSL module; it is returned as SSL output and compared to SSL target (or, if it is None, to the input data),
  • 'contrastive': the input data and its modification pass through the base network feature extraction module and the SSL module; an (input results, modification results) tuple is returned as SSL output,
  • 'contrastive_2layers': the input data and its modification pass through the base network feature extraction module; the output of the second feature extraction layer for the modified data goes through an SSL module and then, optionally, that result and the first-level unmodified features pass another transformation; an (input results, modified results) tuple is returned as SSL output,
